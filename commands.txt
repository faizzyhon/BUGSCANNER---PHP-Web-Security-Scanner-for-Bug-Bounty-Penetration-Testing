  --target https://target.com \
  --i-have-permission

# Same but verbose (shows each finding as it's discovered)

  --i-have-permission \
  --verbose

  --target https://target.com \
  --threads 20 \

# Skip ALL Phase-0 intel (pure vuln scan, fastest)

Step 1: Start the Web GUI for live monitoring
  python web_gui.py

Step 2: In a new terminal — run full recon + vuln scan
  python main.py scan \
    --target https://target.com \
    --threads 20 \
    --timeout 20 \
    --verbose \
    --i-have-permission

Step 3: Review the Intel tab in Web GUI for:
  - Open ports (check for CRITICAL/HIGH risk services)
  - Detected technologies (CMS, server, PHP version)
  - CVEs found for detected software
  - WAF detection (if any, plan bypasses)

Step 4: Run targeted tests based on detected tech
  # If WordPress detected:
  python main.py scan \
    --target https://target.com \
    --modules sqli,xss,lfi,php \
    --i-have-permission

  # If Java/Spring detected:
  python main.py scan \
    --target https://target.com \
    --modules ssrf,xxe,cmdi \
    --i-have-permission

Step 5: Run AI zero-day engine
  python main.py scan \
    --target https://target.com \
    --ai-provider ollama \
    --ai-model deepseek-r1 \
    --ai-only \
    --i-have-permission

Step 6: Review PDF report in ./reports/ directory

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

PHASE-0 INTELLIGENCE (always auto-runs unless --skip-recon):
  recon          IP resolution, DNS, Whois, Geolocation, HTTP headers, WAF
  ports          TCP port scan (70 ports), service banners, risk classification
  tech           Technology fingerprinting (80+ signatures)
  cve            NVD CVE database lookup + active exploit tests

PHASE-1 VULNERABILITY MODULES (--modules):
  sqli           SQL Injection
  xss            Cross-Site Scripting (reflected, stored, DOM)
  ssrf           Server-Side Request Forgery
  lfi            Local File Inclusion / Path Traversal
  xxe            XML External Entity Injection
  cmdi           Command Injection (OS command execution)
  idor           Insecure Direct Object Reference
  jwt            JWT attacks (none algo, weak secret, expired)
  headers        Security headers audit (CSP, HSTS, X-Frame-Options...)
  open_redirect  Open Redirect vulnerabilities
  payment        Payment / balance bypass logic flaws
  php            PHP-specific: DB dumps, file access, webshell upload

SPECIAL:
  all            Run all Phase-1 modules (default)
